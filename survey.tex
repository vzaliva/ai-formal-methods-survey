\section{Scope and structure}
This pack is organised around the two-way intersection between large language models and formal methods:
\emph{formal methods for constraining or checking AI-assisted software development}, and
\emph{LLMs for lowering the cost of formal specification, verification, and mechanised proof}. The report is split into \hyperref[sec:projects]{projects} and \hyperref[sec:papers]{papers}, with separate research and industry subsections, and with an additional classification by technical area. Representative surveys and roadmaps that frame the field include the main autoformalisation survey, the software-requirements survey, the theorem-proving survey, and broader roadmaps on trustworthy AI agents and formal mathematical reasoning.~\cite{weng2025autoformalizationeralargelanguage,beg2025shortsurveyformalisingsoftware,li2024surveydeeplearningtheorem,zhang2024fusionlargelanguagemodels,yang2024formalmathematicalreasoningnew,roadmap_requirements_engineering_llms_2025}

\section{Projects}\label{sec:projects}

\subsection{Research-led projects}

\subsubsection{Specification languages and requirements}\label{subsec:projects-research-spec}
\begin{itemize}
\item \textbf{VERIFAI}. Early project on traceability and verification of natural-language requirements, useful as background for NL-to-spec workflows.~\cite{verifai_project}
\item \textbf{nl2spec}. Interactive translation from unstructured requirements to temporal logic with editable sub-translations.~\cite{cosler2023nl2specinteractivelytranslatingunstructured}
\item \textbf{NL2TL}. A general framework for mapping natural-language instructions to temporal logic across multiple domains.~\cite{chen2024nl2tltransformingnaturallanguages}
\item \textbf{TLAiBench}. A benchmark repository for TLA+ specification tasks.~\cite{tlai_bench_project}
\item \textbf{Doc2Spec}. A multi-agent approach that induces formal specifications from natural-language documents.~\cite{xia2026doc2specsynthesizingformalprogramming}
\item \textbf{Trustworthy formal natural-language specifications}. A proof-assistant-native route to formalising natural-language specifications inside Lean.~\cite{gordon2023trustworthyformalnaturallanguage}
\end{itemize}

\subsubsection{Verified code generation and proof synthesis for code}\label{subsec:projects-research-vcode}
\begin{itemize}
\item \textbf{DafnySynth / dafny-annotator}. LLM-assisted synthesis of Dafny assertions, invariants and proof-oriented annotations.~\cite{poesia2024dafnyannotatoraiassistedverificationdafny}
\item \textbf{AlphaVerus}. Self-improving verified code generation for Verus/Rust.~\cite{alphaverus_project}
\item \textbf{RAG-Verus}. Repository-level proof synthesis for Verus with retrieval augmentation and repository context.~\cite{zhong2025ragverusrepositorylevelprogramverification}
\item \textbf{VeCoGen}. Formally verified C generation using LLMs plus a verifier-guided refinement loop.~\cite{sevenhuijsen2025vecogenautomatinggenerationformally}
\item \textbf{VeriSoftBench}. Repository-scale Lean benchmark focused on software-verification obligations rather than pure mathematics.~\cite{xin2026verisoftbenchrepositoryscaleformalverification}
\end{itemize}

\subsubsection{AI-assisted mechanised proving}\label{subsec:projects-research-mech}
\begin{itemize}
\item \textbf{LeanDojo / ReProver}. Open tooling and datasets for Lean-based proving with retrieval-augmented models.~\cite{yang2023leandojotheoremprovingretrievalaugmented}
\item \textbf{Lean Copilot}. LLM inference inside Lean for tactic suggestion, proof search and premise selection.~\cite{lean_copilot_2025}
\item \textbf{DeepSeek-Prover}. Open Lean prover line with synthetic-data training; the public repository tracks released models.~\cite{xin2024deepseekproveradvancingtheoremproving,deepseek_prover_v15_repo}
\item \textbf{APOLLO}. Lean+LLM collaboration driven by compiler-guided repair.~\cite{apollo_2025}
\item \textbf{AutoRocq}. Agentic Rocq/Coq verification via iterative prover feedback; introduced in Agentic Program Verification.~\cite{tu2026agenticverificationsoftwaresystems}
\item \textbf{Isabellm}. A lightweight Isabelle/HOL prover with both a paper and an open implementation.~\cite{hou2026vibecodingllmpoweredtheorem,llm_isabelle_repo}
\item \textbf{LeanNavigator}. Large-scale generation of additional Lean theorems and proofs from Mathlib4.~\cite{yin2025generatingmillionsleantheorems}
\end{itemize}

\subsubsection{Temporal logic, TLA+, model checking, and hardware formal verification}\label{subsec:projects-research-temp}
\begin{itemize}
\item \textbf{Retrieval-augmented TLAPS proof generation}. LLM-assisted proof generation for TLAPS using retrieved verified proof fragments.~\cite{zhou2025retrievalaugmentedtlapsproofgeneration}
\item \textbf{Language-model-guided TLA+ proof automation}. Hierarchical decomposition for TLA+ proof obligations.~\cite{zhou2026languagemodelguidedtla}
\item \textbf{AssertLLM}. Assertion generation from design specifications and waveforms for hardware verification.~\cite{yan2024assertllmgeneratinghardwareverification}
\item \textbf{FVEval}. A benchmark and evaluation framework for hardware formal-verification tasks.~\cite{kang2024fvevalunderstandinglanguagemodel}
\item \textbf{STELLAR}. Structure-guided retrieval and generation for SystemVerilog assertions.~\cite{rajabi2026stellarstructureguidedllmassertion}
\end{itemize}
