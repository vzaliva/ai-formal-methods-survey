\section{Scope and structure}
This survey is organised around the two-way intersection between large language models and formal methods:
\emph{formal methods for constraining or checking AI-assisted software development}, and
\emph{LLMs for lowering the cost of formal specification, verification, and mechanised proof}. The
survey is split into \hyperref[sec:projects]{projects} and \hyperref[sec:papers]{papers}, with
separate research and industry subsections, and with an additional classification by technical
area. Representative surveys and roadmaps that frame the field include the main autoformalisation
survey, the software-requirements survey, the theorem-proving survey, and broader roadmaps on
trustworthy AI agents and formal mathematical reasoning.~\cite{DBLP:journals/corr/abs-2505-23486,DBLP:journals/corr/abs-2506-11874,DBLP:journals/corr/abs-2404-09939,DBLP:journals/corr/abs-2412-06512,DBLP:conf/icml/YangPHLLCS25,DBLP:journals/infsof/FerrariS25}

\section{Projects}\label{sec:projects}

\subsection{Research-led projects}

\subsubsection{Specification languages and requirements}\label{subsec:projects-research-spec}
\begin{itemize}
\item \textbf{VERIFAI}. Early project on traceability and verification of natural-language requirements, useful as background for NL-to-spec workflows.~\cite{verifai_project}
\item \textbf{nl2spec}. Interactive translation from unstructured requirements to temporal logic with editable sub-translations.~\cite{cosler2023nl2specinteractivelytranslatingunstructured}
\item \textbf{NL2TL}. A general framework for mapping natural-language instructions to temporal logic across multiple domains.~\cite{DBLP:conf/emnlp/ChenGZF23}
\item \textbf{TLAiBench}. A benchmark repository for TLA+ specification tasks.~\cite{tlai_bench_project}
\item \textbf{Doc2Spec}. A multi-agent approach that induces formal specifications from natural-language documents.~\cite{xia2026doc2specsynthesizingformalprogramming}
\item \textbf{Trustworthy formal natural-language specifications}. A proof-assistant-native route to formalising natural-language specifications inside Lean.~\cite{gordon2023trustworthyformalnaturallanguage}
\end{itemize}

\subsubsection{Verified code generation and proof synthesis for code}\label{subsec:projects-research-vcode}
\begin{itemize}
\item \textbf{DafnySynth / dafny-annotator}. LLM-assisted synthesis of Dafny assertions, invariants and proof-oriented annotations.~\cite{DBLP:journals/corr/abs-2411-15143}
\item \textbf{AlphaVerus}. Self-improving verified code generation for Verus/Rust.~\cite{alphaverus_project}
\item \textbf{RAG-Verus}. Repository-level proof synthesis for Verus with retrieval augmentation and repository context.~\cite{DBLP:journals/corr/abs-2502-05344}
\item \textbf{VeCoGen}. Formally verified C generation using LLMs plus a verifier-guided refinement loop.~\cite{DBLP:conf/icse-formalise/SevenhuijsenEN25}
\item \textbf{VeriSoftBench}. Repository-scale Lean benchmark focused on software-verification obligations rather than pure mathematics.~\cite{DBLP:journals/corr/abs-2602-18307}
\end{itemize}

\subsubsection{AI-assisted mechanised proving}\label{subsec:projects-research-mech}
\begin{itemize}
\item \textbf{LeanDojo / ReProver}. Open tooling and datasets for Lean-based proving with retrieval-augmented models.~\cite{DBLP:journals/corr/abs-2306-15626}
\item \textbf{Lean Copilot}. LLM inference inside Lean for tactic suggestion, proof search and premise selection.~\cite{lean_copilot_2025}
\item \textbf{DeepSeek-Prover}. Open Lean prover line with synthetic-data training; the public repository tracks released models.~\cite{DBLP:journals/corr/abs-2405-14333,deepseek_prover_v15_repo}
\item \textbf{APOLLO}. Lean+LLM collaboration driven by compiler-guided repair.~\cite{apollo_2025}
\item \textbf{AutoRocq}. Agentic Rocq/Coq verification via iterative prover feedback; introduced in Agentic Program Verification.~\cite{tu2026agenticverificationsoftwaresystems}
\item \textbf{Isabellm}. A lightweight Isabelle/HOL prover with both a paper and an open implementation.~\cite{hou2026vibecodingllmpoweredtheorem,llm_isabelle_repo}
\item \textbf{LeanNavigator}. Large-scale generation of additional Lean theorems and proofs from Mathlib4.~\cite{DBLP:journals/corr/abs-2503-04772}
\end{itemize}

\subsubsection{Temporal logic, TLA+, model checking, and hardware formal verification}\label{subsec:projects-research-temp}
\begin{itemize}
\item \textbf{Retrieval-augmented TLAPS proof generation}. LLM-assisted proof generation for TLAPS using retrieved verified proof fragments.~\cite{zhou2025retrievalaugmentedtlapsproofgeneration}
\item \textbf{Language-model-guided TLA+ proof automation}. Hierarchical decomposition for TLA+ proof obligations.~\cite{DBLP:journals/corr/abs-2512-09758}
\item \textbf{AssertLLM}. Assertion generation from design specifications and waveforms for hardware verification.~\cite{DBLP:conf/aspdac/0003FLLLXZ25}
\item \textbf{FVEval}. A benchmark and evaluation framework for hardware formal-verification tasks.~\cite{DBLP:journals/corr/abs-2410-23299}
\item \textbf{STELLAR}. Structure-guided retrieval and generation for SystemVerilog assertions.~\cite{DBLP:journals/corr/abs-2601-19903}
\end{itemize}

\subsection{Industry and joint industry--academic projects}
\subsubsection{Specification and intent formalisation}

\begin{itemize}
\item \textbf{Microsoft intent formalisation agenda}. Positioning user-intent formalisation as a core reliability problem for AI-assisted coding.~\cite{intent_formalization_grand_challenge_project}
\item \textbf{nl2postcond}. Microsoft-led work on turning programmer intent into checkable postconditions.~\cite{endres2024largelanguagemodelstransform}
\end{itemize}

\subsubsection{Verified systems code}

\begin{itemize}
\item \textbf{Microsoft verus-proof-synthesis}. Umbrella repository for Microsoft's Verus verification automation stack.~\cite{microsoft_verus_proof_synthesis_repo}
\item \textbf{AutoVerus}. Automated proof generation for Rust/Verus.~\cite{DBLP:journals/pacmpl/YangLMYCGHLLL0Z25}
\item \textbf{VeruSAGE}. Agent-based verification for repository-scale Rust systems.~\cite{DBLP:journals/corr/abs-2512-18436}
\item \textbf{VeriStruct}. Planner-based verification of Verus data-structure modules.~\cite{DBLP:journals/corr/abs-2510-25015}
\end{itemize}

\subsubsection{Formal mathematics and conjectures}

\begin{itemize}
\item \textbf{AlphaProof programme}. Google DeepMind's public project framing around reinforcement learning for formal mathematical reasoning.~\cite{alphaproof_blog_2024}
\item \textbf{Harmonic / Aristotle}. Industry effort on olympiad-level automated theorem proving and formal mathematical reasoning.~\cite{DBLP:journals/corr/abs-2510-01346}
\item \textbf{formal-conjectures}. Repository of Lean-formalised conjectures intended to support formal reasoning research.~\cite{formal_conjectures_repo}
\end{itemize}

\subsubsection{Safety-critical and hardware application}

\begin{itemize}
\item \textbf{Automotive C generation with formal verification}. Joint industry-academic evaluation on safety-critical automotive software.~\cite{safety_critical_automotive_2025}
\item \textbf{AssertLLM}. Joint academic-industry hardware FV workflow.~\cite{DBLP:conf/aspdac/0003FLLLXZ25}
\end{itemize}
